! = head-body kısayol

meta charset= karakter koruması (türkçe karakter konabiliyor)
meta name = genişlik problemini çözüyor(ekran genişliği)
title = başlık (sekme adı)
index = açılan sayfanın adı (arama yapınca çıkan sonuç gibi)
h1 = başlık (/h1 ile kapanır / bize bu imkanı sağlar)
a = Tıklamalı Yazı(link yönlendirme)
p = paragraf oluşturma
lang / href vb = ara birim 
href = link yazmamıza yarıyor (1 özelliği bu)
DEL = Yazının üstünü çizme
b = yazı kalınlaştırma
br = alt satıra geçme
i= eğik yazı
sub = cümledeki küçük şeyler alta yazma h2o daki 2 yi kuralına göre 
sup = cümledeki küçük şeyleri üste yazma 2^2 gibi
img = resim ekleme   src= resim kaynağı     alt = yazı

**Tablo**

table = tablo oluşturma

tr = tablo satırı
td = tablo satır değeri
ul = listelemek aşağı (yukarı)
li = liste elemenaları eklemek (tr,td gibi )
ol = içine li de katınca başınma 1 2 3 ekleme gibi işler yapıyor
th = kalın yazma
colspan= yer kaplicak alan 
border = çerçeve kalınlığı
blockquote = uzun alıntılar için
caption = tablo başlığı
q = kısa alıntılar için
abbr = kısaltmalar için (a daki title gibi)
address = iletişim bilgileri için
cite= kaynak belirtmek için
background = arka plan işleri
bdo = metnin yönünü değiştirmek için
<style>body{}</style> --> body{} = style içerisindeki tanımlı şeyi düzenlemek sülü püslü hale getirmek



style = yazdığım yazının veya herhangi bir şeyin şeklini / biçimini değiştirme (i,b gibi elementlerden farklı)
Örnek Kod
**<h1 style="color: crimson;">Stiller</h1>**  

font-family,font-size  (kısaca style sayesinde bir elementi düzenleyebiliyorum )
